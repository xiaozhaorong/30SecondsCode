01常用的字符串转换
---
- 金额加￥
```
// -100 转换 -￥100
function formatMoney(money) {
  return money < 0 ? `-￥${Math.abs(money)}` : `￥${money}`
}
console.log(formatMoney(-100))
// -￥100
```

- 字符串文字过长 截取换成...
```
// 字符串文字过长 截取换成...
function formatStr(str){
    str = String(str)
    return str.length > 8
        ? str.substr(0,8) + '.'.repeat(3)
        : str  
}
console.log(formatStr(12345678910))
// 12345678
```
